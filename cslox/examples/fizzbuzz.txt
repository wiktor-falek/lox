fun mod(a, n) {
  return a - ( n * int(a / n) );
}

for (var i = 1; i < 16; i = i + 1) {
  var mod3 = mod(i, 3);
  var mod5 = mod(i, 5);

  if (mod3 == 0) {
    if (mod5 == 0) {
      print("FizzBuzz");
    } else {
      print("Fizz");
    }
  } else {
    if (mod5 == 0) {
      print("Buzz");
    } else {
      print(i);
    }
  }
}
