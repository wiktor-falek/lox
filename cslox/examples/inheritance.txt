class Base {
  base() {
    print("base method");
  }

  static static_base() {
    print("static base method");
  }
}

class Derived < Base {
  derived() {
    print("derived method");
  }

  static static_derived() {
    print("static derived method");
  }
}

var base = Base();
var derived = Derived();

Derived.static_derived();
Derived.static_base();

Base.static_base();

derived.derived();
